% Comment command: \com{red text}{blue comment}
% In math mode: adds blue comment in \text{} with spacing
% In text mode: adds blue comment in parentheses right after
\newcommand{\com}[2]{%
	\textcolor{red}{#1}%
	\ifmmode
		\quad\text{\textcolor{blue}{(#2)}}%
	\else
		{\textcolor{blue}{ (#2)}}%
	\fi
}

% Comment environment: for complex content with figures, paragraphs, etc.
% Usage: \begin{comenv}{blue comment} ... content in red ... \end{comenv}
\usepackage{xparse}
\NewDocumentEnvironment{comenv}{+m}{%
	\gdef\comenvarg{#1}%
	\captionsetup{font={color=red}, labelfont={color=red}}%
	\color{red}\ignorespaces
}{%
	\ignorespacesafterend\color{black}%
	\par\noindent{\color{blue}(\comenvarg)}%
}
